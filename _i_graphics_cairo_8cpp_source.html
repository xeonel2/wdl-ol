354 IColor IGraphicsCairo::GetPoint(int x, int y)
355 {
356  // Convert suface to cairo image surface of one pixel (avoid copying the whole surface)
357 
358  cairo_surface_t* pOutSurface = cairo_image_surface_create(CAIRO_FORMAT_ARGB32, 1, 1);
359  cairo_t* pOutContext = cairo_create(pOutSurface);
360  cairo_set_source_surface(pOutContext, mSurface, -x, -y);
361  cairo_paint(pOutContext);
362  cairo_surface_flush(pOutSurface);
363 
364  uint8_t* pData = cairo_image_surface_get_data(pOutSurface);
365  uint32_t px = *((uint32_t*)(pData));
366 
367  cairo_surface_destroy(pOutSurface);
368  cairo_destroy(pOutContext);
369 
370  int A = (px >> 0) & 0xFF;
371  int R = (px >> 8) & 0xFF;
372  int G = (px >> 16) & 0xFF;
373  int B = (px >> 24) & 0xFF;
374 
375  return IColor(A, R, G, B);
