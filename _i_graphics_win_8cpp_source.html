395 LRESULT CALLBACK IGraphicsWin::ParamEditProc(HWND hWnd, UINT msg, WPARAM wParam, LPARAM lParam)
396 {
397  IGraphicsWin* pGraphics = (IGraphicsWin*) GetWindowLongPtr(GetParent(hWnd), GWLP_USERDATA);
398 
399  if (pGraphics && pGraphics->mParamEditWnd && pGraphics->mParamEditWnd == hWnd)
400  {
401  pGraphics->HideTooltip();
402 
403  switch (msg)
404  {
405  case WM_CHAR:
406  {
407  // limit to numbers for text entry on appropriate parameters
408  if(pGraphics->mEdParam)
409  {
410  char c = wParam;
411 
412  if(c == 0x08) break; // backspace
413 
414  switch ( pGraphics->mEdParam->Type() )
415  {
416  case IParam::kTypeEnum:
417  case IParam::kTypeInt:
418  case IParam::kTypeBool:
419  if (c >= '0' && c <= '9') break;
420  else if (c == '-') break;
421  else if (c == '+') break;
422  else return 0;
423  case IParam::kTypeDouble:
424  if (c >= '0' && c <= '9') break;
425  else if (c == '-') break;
426  else if (c == '+') break;
427  else if (c == '.') break;
428  else return 0;
429  default:
430  break;
431  }
432  }
433  break;
434  }
435  case WM_KEYDOWN:
436  {
437  if (wParam == VK_RETURN)
438  {
439  pGraphics->mParamEditMsg = kCommit;
440  return 0;
441  }
442  else if (wParam == VK_ESCAPE)
443  {
444  pGraphics->mParamEditMsg = kCancel;
445  return 0;
446  }
447  break;
448  }
449  case WM_SETFOCUS:
450  {
451  pGraphics->mParamEditMsg = kEditing;
452  break;
453  }
454  case WM_KILLFOCUS:
455  {
456  pGraphics->mParamEditMsg = kCommit;
457  break;
458  }
459  // handle WM_GETDLGCODE so that we can say that we want the return key message
460  // (normally single line edit boxes don't get sent return key messages)
461  case WM_GETDLGCODE:
462  {
463  if (pGraphics->mEdParam) break;
464  LPARAM lres;
465  // find out if the original control wants it
466  lres = CallWindowProc(pGraphics->mDefEditProc, hWnd, WM_GETDLGCODE, wParam, lParam);
467  // add in that we want it if it is a return keydown
468  if (lParam && ((MSG*)lParam)->message == WM_KEYDOWN && wParam == VK_RETURN)
469  {
470  lres |= DLGC_WANTMESSAGE;
471  }
472  return lres;
473  }
474  case WM_COMMAND:
475  {
476  switch HIWORD(wParam)
477  {
478  case CBN_SELCHANGE:
479  {
480  if (pGraphics->mParamEditWnd)
481  {
482  pGraphics->mParamEditMsg = kCommit;
483  return 0;
484  }
485  }
486 
487  }
488  break; // Else let the default proc handle it.
489  }
490  }
491  return CallWindowProc(pGraphics->mDefEditProc, hWnd, msg, wParam, lParam);
492  }
493  return DefWindowProc(hWnd, msg, wParam, lParam);
494 }
