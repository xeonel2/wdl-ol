255 {

311 void IGraphics::DrawBitmapedText(IBitmap& bitmap, IRECT& rect, IText& text, IBlend* pBlend, const char* str, bool vCenter, bool multiline, int charWidth, int charHeight, int charOffset)
312 {
313  if (CSTR_NOT_EMPTY(str))
314  {
315  int stringLength = (int) strlen(str);
316 
317  float basicYOffset = 0.;
318  float basicXOffset = 0.;
319 
320  if (vCenter)
321  basicYOffset = rect.T + ((rect.H() - charHeight) / 2.f);
322  else
323  basicYOffset = rect.T;
324 
325  if (text.mAlign == IText::kAlignCenter)
326  basicXOffset = rect.L + ((rect.W() - (stringLength * charWidth)) / 2.f);
327  else if (text.mAlign == IText::kAlignNear)
328  basicXOffset = rect.L;
329  else if (text.mAlign == IText::kAlignFar)
330  basicXOffset = rect.R - (stringLength * charWidth);
331 
332  int widthAsOneLine = charWidth * stringLength;
333 
334  int nLines;
335  int stridx = 0;
336 
337  int nCharsThatFitIntoLine;
338 
339  if(multiline)
340  {
341  if (widthAsOneLine > rect.W())
342  {
343  nCharsThatFitIntoLine = int(rect.W() / (float)charWidth);
344  nLines = int(float(widthAsOneLine) / rect.W()) + 1;
345  }
346  else // line is shorter than width of rect
347  {
348  nCharsThatFitIntoLine = stringLength;
349  nLines = 1;
350  }
351  }
352  else
353  {
354  nCharsThatFitIntoLine = int(rect.W() / (float) charWidth);
355  nLines = 1;
356  }
357 
358  for(int line=0; line<nLines; line++)
359  {
360  float yOffset = basicYOffset + line * charHeight;
361 
362  for(int linepos=0; linepos<nCharsThatFitIntoLine; linepos++)
363  {
364  if (str[stridx] == '\0') return;
365 
366  int frameOffset = (int) str[stridx++] - 31; // calculate which frame to look up
367 
368  float xOffset = ((float) linepos * ((float) charWidth + (float) charOffset)) + basicXOffset; // calculate xOffset for character we're drawing
369  IRECT charRect = IRECT(xOffset, yOffset, xOffset + charWidth, yOffset + charHeight);
370  DrawBitmap(bitmap, charRect, frameOffset, pBlend);
371  }
372  }
373  }
374 }

643  return;
644  }
645  #endif
646

650  }

653  }

1006 IBitmap IGraphics::LoadBitmap(const char* name, int nStates, bool framesAreHorizontal)
1007 {
1008  const int targetScale = round(GetDisplayScale());
1009 
1010  APIBitmap* pAPIBitmap = s_bitmapCache.Find(name, targetScale);
1011 
1012  // If the bitmap is not already cached at the targetScale
1013  if (!pAPIBitmap)
1014  {
1015  WDL_String fullPath;
1016  int sourceScale = 0;
1017  bool fromDisk = false;
1018 
1019  if (!SearchImageResource(name, "png", fullPath, targetScale, sourceScale))
1020  {
1021  // If no resource exists then search the cache for a suitable match
1022  pAPIBitmap = SearchBitmapInCache(name, targetScale, sourceScale);
1023  }
1024  else
1025  {
1026  // Try again in cache for mismatched bitmaps, but load from disk if needed
1027  if (sourceScale != targetScale)
1028  pAPIBitmap = s_bitmapCache.Find(name, sourceScale);
1029 
1030  if (!pAPIBitmap)
1031  {
1032  pAPIBitmap = LoadAPIBitmap(fullPath, sourceScale);
1033  fromDisk = true;
1034  }
1035  }
1036 
1037  // Protection from searching for non-existant bitmaps (e.g. typos in config.h or .rc)
1038  assert(pAPIBitmap);
1039 
1040  const IBitmap bitmap(pAPIBitmap, nStates, framesAreHorizontal, name);
1041 
1042  // Scale if needed
1043  if (pAPIBitmap->GetScale() != targetScale)
1044  {
1045  // Scaling adds to the cache but if we've loaded from disk then we need to dispose of the temporary APIBitmap
1046  IBitmap scaledBitmap = ScaleBitmap(bitmap, name, targetScale);
1047  if (fromDisk)
1048  delete pAPIBitmap;
1049  return scaledBitmap;
1050  }
1051 
1052  // Retain if we've newly loaded from disk
1053  if (fromDisk)
1054  RetainBitmap(bitmap, name);
1055  }
1056 
1057  return IBitmap(pAPIBitmap, nStates, framesAreHorizontal, name);
1058 }
