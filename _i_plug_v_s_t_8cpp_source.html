86 EHost IPlugVST::GetHost()
87 {
88  EHost host = IPLUG_BASE_CLASS::GetHost();
89 
90  if (host == kHostUninit)
91  {
92  char productStr[256];
93  productStr[0] = '\0';
94  int version = 0;
95  mHostCallback(&mAEffect, audioMasterGetProductString, 0, 0, productStr, 0.0f);
96 
97  if (CSTR_NOT_EMPTY(productStr))
98  {
99  int decVer = (int) mHostCallback(&mAEffect, audioMasterGetVendorVersion, 0, 0, 0, 0.0f);
100  int ver = decVer / 10000;
101  int rmaj = (decVer - 10000 * ver) / 100;
102  int rmin = (decVer - 10000 * ver - 100 * rmaj);
103  version = (ver << 16) + (rmaj << 8) + rmin;
104  }
105 
106  SetHost(productStr, version);
107  host = IPLUG_BASE_CLASS::GetHost();
108  }
109 
110  return host;
111 }
