413  dest.attach(pCopy->buf(), pCopy->width(), pCopy->height(), pCopy->row_bytes());

423  double ratioH = (double) destH / pSourcePixelMap->height();

461 void IGraphicsAGG::CalculateTextLines(WDL_TypedBuf<LineInfo>* pLines, const IRECT& rect, const char* str, FontManagerType& manager)
462 {
463  LineInfo info;
464  info.mStartChar = 0;
465  info.mEndChar = (int) strlen(str);
466  pLines->Add(info);
467 
468  LineInfo* pLine = pLines->Get();
469 
470  size_t lineStart = 0;
471  size_t lineWidth = 0;
472  size_t linePos = 0;
473  double xCount = 0.0;
474 
475  const char* cstr = str;
476 
477  while (*cstr)
478  {
479  const agg::glyph_cache* pGlyph = manager.glyph(*cstr);
480 
481  if (pGlyph)
482  {
483  xCount += pGlyph->advance_x;
484  }
485 
486  cstr++;
487  linePos++;
488 
489  if (*cstr == ' ' || *cstr == 0)
490  {
491  pLine->mStartChar = (int) lineStart;
492  pLine->mEndChar = (int) linePos;
493  pLine->mWidth = xCount;
494  }
495 
496  if (rect.W() > 0 && xCount >= rect.W())
497  {
498  assert(pLine);
499 
500  cstr = &str[pLine->mEndChar];
501  lineStart = pLine->mEndChar + 1;
502  linePos = pLine->mEndChar;
503 
504  LineInfo info;
505  pLines->Add(info);
506  pLine++;
507 
508  assert(pLines);
509 
510  xCount = 0;
511  lineWidth = 0;
512  }
513 
514  }
515 }
